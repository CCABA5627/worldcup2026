const SHEET_NAME = "submissions"; const POINTS = { "group": 3, "draw": 1, "round32": 5, "round16": 8, "quarter": 13, "semi": 20, "champion": 35 }; function doGet(e) { const action = e.parameter.action; if (action === "submit") { return submitEntry(e.parameter, e.parameter.callback); } if (action === "leaderboard") { return getLeaderboard(e.parameter.callback); } if (action === "updatePoints") { const updateData = { stage: e.parameter.stage, teams: JSON.parse(e.parameter.teams || "[]"), undo: e.parameter.undo === "true" }; return updatePoints(updateData, e.parameter.callback); } return makeResponse({ error: "Unknown action" }); } function doPost(e) { try { let data; if (e.postData && e.postData.contents) { data = JSON.parse(e.postData.contents); } else { return makeResponse({ success: false, error: "No data received." }); } if (data.action === "updatePoints") { return updatePoints(data); } return processSubmission(data, callback); } catch (err) { return makeResponse({ success: false, error: err.message }); } } function submitEntry(params, callback) { try { const data = { firstName: params.firstName || "", lastName: params.lastName || "", email: params.email || "", teamName: params.teamName || "", teams: JSON.parse(params.teams || "[]"), entryFee: parseFloat(params.entryFee) || 10 }; return processSubmission(data, callback); } catch(err) { return makeResponse({ success: false, error: err.message }); } } function processSubmission(data, callback) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const existing = sheet.getDataRange().getValues(); const emailRows = existing.filter(r => r[3] === data.email); if (emailRows.length > 0 && !data.teamName) { return makeResponse({ success: false, error: "A team name is required for multiple entries." }, callback); } sheet.appendRow([ new Date().toISOString(), data.firstName, data.lastName, data.email, data.teamName || "", data.teams[0] || "", data.teams[1] || "", data.teams[2] || "", data.teams[3] || "", data.teams[4] || "", data.entryFee || 5, 0, "" ]); return makeResponse({ success: true }, callback); } function updatePoints(data, callback) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const pointsPerStage = POINTS[data.stage] || 0; const multiplier = data.undo ? -1 : 1; const winningTeams = data.teams || []; let updatedCount = 0; for (let i = 1; i < rows.length; i++) { const playerTeams = [rows[i][5], rows[i][6], rows[i][7], rows[i][8], rows[i][9]].filter(Boolean); let pointsToAdd = 0; playerTeams.forEach(function(t) { if (winningTeams.indexOf(t) !== -1) pointsToAdd += pointsPerStage; }); if (pointsToAdd > 0) { const currentPoints = parseInt(rows[i][11]) || 0; const newPoints = Math.max(0, currentPoints + (pointsToAdd * multiplier)); sheet.getRange(i + 1, 12).setValue(newPoints); updatedCount++; } } return makeResponse({ success: true, updated: updatedCount, pointsPerTeam: pointsPerStage }, callback); } catch (err) { return makeResponse({ success: false, error: err.message }, callback); } } function getLeaderboard(callback) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); if (data.length <= 1) { return makeResponse({ entries: [], totalPaid: 0 }, callback); } const entries = []; let totalPaid = 0; for (let i = 1; i < data.length; i++) { const row = data[i]; const teams = [row[5], row[6], row[7], row[8], row[9]].filter(Boolean); const fee = parseFloat(row[10]) || 0; totalPaid += fee; entries.push({ firstName: row[1], lastName: row[2], teamName: row[4], teams: teams, entryFee: fee, points: parseInt(row[11]) || 0, paid: (row[12] || "").toString().toUpperCase() === "YES" }); } entries.sort((a, b) => b.points - a.points); return makeResponse({ entries: entries, totalPaid: totalPaid }, callback); } function makeResponse(data, callback) { if (callback) { return ContentService .createTextOutput(callback + '(' + JSON.stringify(data) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
